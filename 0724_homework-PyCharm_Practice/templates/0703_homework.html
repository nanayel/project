def divisor(num):
    div = []
    for i in range(1, num):
        if num%i == 0:
            div.append(i)
    return div

def perfect_num(num):
    div = divisor(num)
    if num == sum(div):
        return True
    else:
        return False
        
# main
for i in range(1, 10001):
    if perfect_num(i):

Exercise 113 : Formatting a List¶

def formatting_list(words):
    output_str = ''
    if len(words) == 0:
        return 0
    elif len(words) == 1:
        output_str = words[0]
    elif len(words) == 2:
        output_str = words[0] + ' and ' + words[1]
    else:
        output_str = words[0]
        for i in range(1, len(words)-1):
            output_str += ', '
            output_str += words[i]
        output_str += ' and '
        output_str += words[len(words)-1]
 
    return output_str
        
# main
print(formatting_list(['apples']))
print(formatting_list(['apples', 'oranges']))
print(formatting_list(['apples', 'oranges', 'bananas']))
print(formatting_list(['apples', 'oranges', 'bananas', 'lemons']))
