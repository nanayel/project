Exercise 66 : Compute a Grade Point Average¶

def grade(letter):
    if letter == 'A+':
        return 4.0
    elif letter == 'A':
        return 4.0
    elif letter == 'A-':
        return 3.7
    elif letter == 'B+':
        return 3.3
    elif letter == 'B':
        return 3.0
    elif letter == 'B-':
        return 2.7
    if letter == 'C+':
        return 2.3
    elif letter == 'C':
        return 2.0
    elif letter == 'C-':
        return 1.7
    elif letter == 'D+':
        return 1.3
    elif letter == 'D':
        return 1.0
    elif letter == 'F':
        return 0.0
    else:
        return False

# main
import numpy

letter = input()
letters = []

while letter:
    if grade(letter):
        letters.append(grade(letter))
    else:
        print('Plese enter a appropriate letter')
    letter = input()

print(numpy.mean(letters))

def price_admission(age):
    if age <= 2:
        return 0.0
    elif 3 <= age <= 12:
        return 14.0
    elif age >= 65:
        return 18.0
    else:
        return 23.0

# main
line = input()
nums = []

while line:
    try:
        num = int(line)
    except:
        print('Please enter an age.')
    else:
        nums.append(price_admission(num))
    line = input()

print("{:.2f}".format(sum(nums)))

def GCD(d, num1, num2):
    while True:
        if num1%d == 0 and num2%d == 0:
            return d
        else:
            d -= 1

# main
line = input()
nums = []

while line != 'q':
    nums = line.split(',')
    try:
        num1 = int(nums[0])
        num2 = int(nums[1])
    except:
        print('Please enter an integer.')
    else:
        if num1 < num2:
            d = num1
            gcd = GCD(d, num1, num2)
        elif num1 == num2:
            gcd = num1
        else:
            d = num2
            gcd = GCD(d, num1, num2)
        print(gcd)
    line = input()

def bin_to_dec(nums):
    for i in range(len(nums)):
        nums[i] = int(nums[i]) * pow(2, len(nums)-1-i)
    return sum(nums)

# main
print('if you want to quit, press \'q\'')
line = input()
nums = []

while line != 'q':
    try:
        num = int(line)
    except:
        print('Please enter a binary number')
    else:
        nums = list(line)
        print(num, end='')
        print('(2) = ', end='')
        print(bin_to_dec(nums), end='')
        print('(10)')
    line = input()

def dec_to_bin(num):
    string = ''
    while num != 0:
        r = num%2
        string = str(r) + string
        num = int(num/2)
    return string

# main
print('if you want to quit, press \'q\'')
line = input()

while line != 'q':
    try:
        num = int(line)

        print('Please enter a binary number')
    else:

        print('(10) = ', end='')
        print(dec_to_bin(num), end='')
        print('(2)')
