def capitalize(line):
    line[0] = line[0].upper()
    for i in range(len(line)):
        if line[i] == ' ':
            if i+1 < len(line) and line[i+1] == 'i':
                line[i+1] = 'I'
        elif line[i] == '"':
            if i+1 < len(line):
                line[i+1] = line[i+1].upper()
        elif line[i] == '?' or line[i] == '!' or line[i] == '.':
            if i+2 < len(line):
                line[i+2] = line[i+2].upper()
        else:
            continue
    return ''.join(line)

# main
line_ = input()

while line_:
    print(capitalize(list(line_)))
    line_ = input()

        return False
    else:

        if i == int(num/2):

def prime(num):
    if num == 1:
        return False
    elif num == 2:
        return True
    else:
        for i in range(2, int(num/2 + 1)):
            if num%i == 0:
                return False
        if i == int(num/2):
            return True

def nextPrime(num):
    j = num+1
    while True:
        if prime(j):
            return j
        j+=1

# main
line = input()

while line != 'q':

        num = int(line)

def good_password(line):
    count1 = count2 = count3 = 0
    if len(line) < 8:
        return False

            count1=1

            count2=1
        elif line[i].isnumeric():
            count3=1

    if count1+count2+count3 == 3:
        return True

        return False
