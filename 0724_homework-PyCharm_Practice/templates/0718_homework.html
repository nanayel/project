def find_drawing(artist_name):
    url = "https://www.pinterest.co.kr/search/pins/?q="
    url += artist_name

    # 각자의 webdriver의 위치 : 여기서는 chrome webdriver 이용
    browser = webdriver.Chrome('/Users/sony/Downloads/chromedriver_win32/chromedriver')
    browser.get(url)
    time.sleep(1)

    element = browser.find_element_by_tag_name("body")

    no_of_pagedowns = 50
    while no_of_pagedowns:
        element.send_keys(Keys.PAGE_DOWN)
        time.sleep(0.3)
        no_of_pagedowns -= 1
        try:
            browser.execute_script("window.scrollTo(0, element.scrollHeight")
        except:
            None

    html = browser.page_source
    dom = BeautifulSoup(html, "lxml")
    links = dom.select("div img")

    image_link = []
    for link in links:
        if link.has_attr("src"):
            image_link.append(link["src"])

    return image_link

import base64
import requests

def drawing_bytes(image_list):
    output = []
    for link in image_list:
        data = requests.get(link)
        data_bytes = base64.encodebytes(data.content)
        data_list = list(str(data_bytes))
        data_list = data_list[2:len(data_list)-3]
        data_bytes = "".join(data_list)
        output.append(data_bytes)

    return output
